444  std::string help(size_t wid=30, std::string prev="") const {
445  // Delegate to subcommand if needed
446  if(prev == "")
447  prev = name;
448  else
449  prev += " " + name;
450 
451  if(selected_subcommands.size() > 0)
452  return selected_subcommands.at(0)->help(wid, prev);
453 
454  std::stringstream out;
455  out << description << std::endl;
456  out << "Usage: " << prev;
457 
458  // Check for options
459  bool npos = false;
460  std::set<std::string> groups;
461  for(const Option_p &opt : options) {
462  if(opt->nonpositional()) {
463  npos = true;
464  groups.insert(opt->get_group());
465  }
466  }
467 
468  if(npos)
469  out << " [OPTIONS]";
470 
471  // Positionals
472  bool pos=false;
473  for(const Option_p &opt : options)
474  if(opt->get_positional()) {
475  // A hidden positional should still show up in the usage statement
476  //if(detail::to_lower(opt->get_group()) == "hidden")
477  // continue;
478  out << " " << opt->help_positional();
479  if(opt->has_description())
480  pos=true;
481  }
482 
483  if(subcommands.size() > 0) {
484  if(required_subcommand != 0)
485  out << " SUBCOMMAND";
486  else
487  out << " [SUBCOMMAND]";
488  }
489 
490  out << std::endl << std::endl;
491 
492  // Positional descriptions
493  if(pos) {
494  out << "Positionals:" << std::endl;
495  for(const Option_p &opt : options) {
496  if(detail::to_lower(opt->get_group()) == "hidden")
497  continue;
498  if(opt->get_positional() && opt->has_description())
499  detail::format_help(out, opt->help_pname(), opt->get_description(), wid);
500  }
501  out << std::endl;
502 
503  }
504 
505 
506  // Options
507  if(npos) {
508  for (const std::string& group : groups) {
509  if(detail::to_lower(group) == "hidden")
510  continue;
511  out << group << ":" << std::endl;
512  for(const Option_p &opt : options) {
513  if(opt->nonpositional() && opt->get_group() == group)
514  detail::format_help(out, opt->help_name(), opt->get_description(), wid);
515 
516  }
517  out << std::endl;
518  }
519  }
520 
521  // Subcommands
522  if(subcommands.size()> 0) {
523  out << "Subcommands:" << std::endl;
524  for(const App_p &com : subcommands)
525  detail::format_help(out, com->get_name(), com->description, wid);
526  }
527  return out.str();
528  }

592  detail::Classifer _recognize(std::string current) const {
593  std::string dummy1, dummy2;
594 
595  if(current == "--")
596  return detail::Classifer::POSITIONAL_MARK;
597  for(const App_p &com : subcommands) {
598  if(com->check_name(current))
599  return detail::Classifer::SUBCOMMAND;
600  }
601  if(detail::split_long(current, dummy1, dummy2))
602  return detail::Classifer::LONG;
603  if(detail::split_short(current, dummy1, dummy2))
604  return detail::Classifer::SHORT;
605  return detail::Classifer::NONE;
606  }

610  std::vector<std::string>& _parse(std::vector<std::string> &args) {
611  parsed = true;
612 
613  bool positional_only = false;
614 
615  while(args.size()>0) {
616 
617 
618  detail::Classifer classifer = positional_only ? detail::Classifer::NONE : _recognize(args.back());
619  switch(classifer) {
620  case detail::Classifer::POSITIONAL_MARK:
621  missing.emplace_back(classifer, args.back());
622  args.pop_back();
623  positional_only = true;
624  break;
625  case detail::Classifer::SUBCOMMAND:
626  _parse_subcommand(args);
627  break;
628  case detail::Classifer::LONG:
629  // If already parsed a subcommand, don't accept options
630  if(selected_subcommands.size() > 0) {
631  missing.emplace_back(classifer, args.back());
632  args.pop_back();
633  } else
634  _parse_long(args);
635  break;
636  case detail::Classifer::SHORT:
637  // If already parsed a subcommand, don't accept options
638  if(selected_subcommands.size() > 0) {
639  missing.emplace_back(classifer, args.back());
640  args.pop_back();
641  } else
642  _parse_short(args);
643  break;
644  case detail::Classifer::NONE:
645  // Probably a positional or something for a parent (sub)command
646  missing.emplace_back(classifer, args.back());
647  args.pop_back();
648  }
649  }
650 
651  if (help_flag != nullptr && help_flag->count() > 0) {
652  throw CallForHelp();
653  }
654 
655 
656  // Collect positionals
657 
658  // Loop over all positionals
659  for(size_t i=0; i<missing.size(); i++) {
660 
661  // Skip non-positionals (speedup)
662  if(missing.at(i).first != detail::Classifer::NONE)
663  continue;
664 
665  // Loop over all options
666  for(const Option_p& opt : options) {
667 
668  // Eat options, one by one, until done
669  while ( opt->get_positional()
670  && opt->count() < opt->get_expected()
671  && i < missing.size()
672  ) {
673 
674  // Skip options, only eat positionals
675  if(missing.at(i).first != detail::Classifer::NONE) {
676  i++;
677  continue;
678  }
679 
680  opt->get_new();
681  opt->add_result(0, missing.at(i).second);
682  missing.erase(missing.begin() + i); // Remove option that was eaten
683  // Don't need to remove 1 from i since this while loop keeps reading i
684  }
685  }
686  }
687 
688  // Process an INI file
689  if (ini_setting != nullptr && ini_file != "") {
690  try {
691  std::vector<std::string> values = detail::parse_ini(ini_file);
692  while(values.size() > 0) {
693  _parse_long(values, false);
694  }
695 
696  } catch (const FileError &) {
697  if(ini_required)
698  throw;
699  }
700  }
701 
702 
703  // Get envname options if not yet passed
704  for(const Option_p& opt : options) {
705  if (opt->count() == 0 && opt->_envname != "") {
706  char *ename = std::getenv(opt->_envname.c_str());
707  if(ename != nullptr) {
708  opt->get_new();
709  opt->add_result(0, std::string(ename));
710  }
711  }
712  }
713 
714  // Process callbacks
715  for(const Option_p& opt : options) {
716  if (opt->count() > 0) {
717  opt->run_callback();
718  }
719  }
720 
721  // Verify required options
722  for(const Option_p& opt : options) {
723  // Required
724  if (opt->get_required()
725  && (opt->count() < opt->get_expected() || opt->count() == 0))
726  throw RequiredError(opt->get_name());
727  // Requires
728  for (const Option* opt_req : opt->_requires)
729  if (opt->count() > 0 && opt_req->count() == 0)
730  throw RequiresError(opt->get_name(), opt_req->get_name());
731  // Excludes
732  for (const Option* opt_ex : opt->_excludes)
733  if (opt->count() > 0 && opt_ex->count() != 0)
734  throw ExcludesError(opt->get_name(), opt_ex->get_name());
735  }
736 
737  if(required_subcommand < 0 && selected_subcommands.size() == 0)
738  throw RequiredError("Subcommand required");
739  else if(required_subcommand > 0 && selected_subcommands.size() != required_subcommand)
740  throw RequiredError(std::to_string(required_subcommand) + " subcommand(s) required");
741 
742  // Convert missing (pairs) to extras (string only)
743  args.resize(missing.size());
744  std::transform(std::begin(missing), std::end(missing), std::begin(args),
745  [](const std::pair<detail::Classifer, std::string>& val){return val.second;});
746  std::reverse(std::begin(args), std::end(args));
747 
748  size_t num_left_over = std::count_if(std::begin(missing), std::end(missing),
749  [](std::pair<detail::Classifer, std::string>& val){return val.first != detail::Classifer::POSITIONAL_MARK;});
750 
751  if(num_left_over>0 && no_extras)
752  throw ExtrasError("[" + detail::join(args, " ") + "]");
753 
754  pre_callback();
755  run_callback();
756 
757  return args;
758  }

775  void _parse_short(std::vector<std::string> &args) {
776  std::string current = args.back();
777 
778  std::string name_;
779  std::string rest;
780  if(!detail::split_short(current, name_, rest))
781  throw HorribleError("Short");
782  args.pop_back();
783 
784  auto op_ptr = std::find_if(std::begin(options), std::end(options), [name_](const Option_p &opt){return opt->check_sname(name_);});
785 
786  if(op_ptr == std::end(options)) {
787  missing.emplace_back(detail::Classifer::SHORT, "-" + name_);
788  return;
789  }
790 
791  // Get a reference to the pointer to make syntax bearable
792  Option_p& op = *op_ptr;
793 
794  int vnum = op->get_new();
795  int num = op->get_expected();
796 
797  if(num == 0)
798  op->add_result(vnum, "");
799  else if(rest!="") {
800  if (num > 0)
801  num--;
802  op->add_result(vnum, rest);
803  rest = "";
804  }
805 
806 
807  if(num == -1) {
808  while(args.size()>0 && _recognize(args.back()) == detail::Classifer::NONE) {
809  op->add_result(vnum, args.back());
810  args.pop_back();
811  }
812  } else while(num>0 && args.size() > 0) {
813  num--;
814  std::string current_ = args.back();
815  args.pop_back();
816  op->add_result(vnum, current_);
817  }
