1281  parsed_ = true;

1284  while(!args.empty()) {
1285  _parse_single(args, positional_only);
1286  }
1287 
1288  for(const Option_p &opt : options_)
1289  if(opt->get_short_circuit() && opt->count() > 0)
1290  opt->run_callback();
1291 
1292  // Process an INI file
1293  if(config_ptr_ != nullptr) {
1294  if(*config_ptr_) {
1295  config_ptr_->run_callback();

1297  }
1298  if(!config_name_.empty()) {
1299  try {
1300  std::vector<ConfigItem> values = config_formatter_->from_file(config_name_);
1301  _parse_config(values);
1302  } catch(const FileError &) {
1303  if(config_required_)
1304  throw;
1305  }
1306  }
1307  }
1308 
1309  // Get envname options if not yet passed
1310  for(const Option_p &opt : options_) {
1311  if(opt->count() == 0 && !opt->envname_.empty()) {
1312  char *buffer = nullptr;
1313  std::string ename_string;
1314 
1315 #ifdef _MSC_VER
1316  // Windows version
1317  size_t sz = 0;
1318  if(_dupenv_s(&buffer, &sz, opt->envname_.c_str()) == 0 && buffer != nullptr) {
1319  ename_string = std::string(buffer);
1320  free(buffer);
1321  }
1322 #else
1323  // This also works on Windows, but gives a warning
1324  buffer = std::getenv(opt->envname_.c_str());
1325  if(buffer != nullptr)
1326  ename_string = std::string(buffer);
1327 #endif
1328 
1329  if(!ename_string.empty()) {
1330  opt->add_result(ename_string);
1331  }
1332  }
1333  }
1334 
1335  // Process callbacks
1336  for(const Option_p &opt : options_) {
1337  if(opt->count() > 0 && !opt->get_callback_run()) {
1338  opt->run_callback();
1339  }
1340  }
1341 
1342  // Verify required options
1343  for(const Option_p &opt : options_) {
1344  // Exit if a help flag was passed (requirements not required in that case)
1345  if(_any_help_flag())
1346  break;
1347 
1348  // Required or partially filled
1349  if(opt->get_required() || opt->count() != 0) {
1350  // Make sure enough -N arguments parsed (+N is already handled in parsing function)
1351  if(opt->get_items_expected() < 0 && opt->count() < static_cast<size_t>(-opt->get_items_expected()))
1352  throw ArgumentMismatch::AtLeast(opt->get_name(), -opt->get_items_expected());
1353 
1354  // Required but empty
1355  if(opt->get_required() && opt->count() == 0)
1356  throw RequiredError(opt->get_name());
1357  }
1358  // Requires
1359  for(const Option *opt_req : opt->requires_)
1360  if(opt->count() > 0 && opt_req->count() == 0)
1361  throw RequiresError(opt->get_name(), opt_req->get_name());
1362  // Excludes
1363  for(const Option *opt_ex : opt->excludes_)
1364  if(opt->count() > 0 && opt_ex->count() != 0)
1365  throw ExcludesError(opt->get_name(), opt_ex->get_name());
1366  }
1367 
1368  auto selected_subcommands = get_subcommands();
1369  if(require_subcommand_min_ > selected_subcommands.size())
1370  throw RequiredError::Subcommand(require_subcommand_min_);
1371 
1372  // Convert missing (pairs) to extras (string only)
1373  if(!(allow_extras_ || prefix_command_)) {
1374  size_t num_left_over = remaining_size();
1375  if(num_left_over > 0) {
1376  args = remaining(false);
1377  throw ExtrasError(args);
1378  }

1383  }
1384  }

1615

1619  }
1620  op->add_result(args.back());

1622  args.pop_back();
1623  collected++;
1624  }
1625

1628  args.pop_back();
1629 
1630  } else {

1632  num--;

1636  parse_order_.push_back(op.get());

1641  }
1642  }
1643 
1644  if(!rest.empty()) {
1645  rest = "-" + rest;
1646  args.push_back(rest);
1647  }
1648  }
