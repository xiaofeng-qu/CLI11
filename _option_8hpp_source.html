528  std::string get_name(bool positional = false, //<[input] Show the positional name
529  bool all_options = false //<[input] Show every option
530  ) const {
531 
532  if(all_options) {
533 
534  std::vector<std::string> name_list;
535 
537  if((positional && pname_.length()) || (snames_.empty() && lnames_.empty()))
538  name_list.push_back(pname_);
539 
540  for(const std::string &sname : snames_)
541  name_list.push_back("-" + sname);
542 
543  for(const std::string &lname : lnames_)
544  name_list.push_back("--" + lname);
545 
546  return detail::join(name_list);
547 
548  } else {
549 
550  // This returns the positional name no matter what
551  if(positional)
552  return pname_;
553 
554  // Prefer long name
555  else if(!lnames_.empty())
556  return std::string("--") + lnames_[0];
557 
558  // Or short name if no long name
559  else if(!snames_.empty())
560  return std::string("-") + snames_[0];
561 
562  // If positional is the only name, it's okay to use that
563  else
564  return pname_;
565  }

573  void run_callback() {
574 
575  callback_run_ = true;
576 
577  // Run the validators (can change the string)
578  if(!validators_.empty()) {
579  for(std::string &result : results_)
580  for(const std::function<std::string(std::string &)> &vali : validators_) {
581  std::string err_msg = vali(result);
582  if(!err_msg.empty())
583  throw ValidationError(get_name(), err_msg);
584  }
585  }
586 
587  bool local_result;
588 
589  // Num items expected or length of vector, always at least 1
590  // Only valid for a trimming policy
591  int trim_size =
592  std::min<int>(std::max<int>(std::abs(get_items_expected()), 1), static_cast<int>(results_.size()));
593 
594  // Operation depends on the policy setting
595  if(multi_option_policy_ == MultiOptionPolicy::TakeLast) {
596  // Allow multi-option sizes (including 0)
597  results_t partial_result{results_.end() - trim_size, results_.end()};
598  local_result = !callback_(partial_result);
599 
600  } else if(multi_option_policy_ == MultiOptionPolicy::TakeFirst) {
601  results_t partial_result{results_.begin(), results_.begin() + trim_size};
602  local_result = !callback_(partial_result);
603 
604  } else if(multi_option_policy_ == MultiOptionPolicy::Join) {
605  results_t partial_result = {detail::join(results_, "\n")};
606  local_result = !callback_(partial_result);
607 
608  } else {
609  // Exact number required
610  if(get_items_expected() > 0) {
611  if(results_.size() != static_cast<size_t>(get_items_expected()))
612  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
613  // Variable length list
614  } else if(get_items_expected() < 0) {
615  // Require that this be a multiple of expected size and at least as many as expected
616  if(results_.size() < static_cast<size_t>(-get_items_expected()) ||
617  results_.size() % static_cast<size_t>(std::abs(get_type_size())) != 0)
618  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
619  }
620  local_result = !callback_(results_);
621  }
622 
623  if(local_result)
624  throw ConversionError(get_name(), results_);
625  }

628  bool operator==(const Option &other) const {
629  for(const std::string &sname : snames_)
630  if(other.check_sname(sname))
631  return true;
632  for(const std::string &lname : lnames_)
633  if(other.check_lname(lname))

635  // We need to do the inverse, just in case we are ignore_case or ignore underscore
636  for(const std::string &sname : other.snames_)
637  if(check_sname(sname))
638  return true;
639  for(const std::string &lname : other.lnames_)
640  if(check_lname(lname))
641  return true;
642  return false;
643  }

646  bool check_name(std::string name) const {
647 
648  if(name.length() > 2 && name.substr(0, 2) == "--")
649  return check_lname(name.substr(2));
650  else if(name.length() > 1 && name.substr(0, 1) == "-")
651  return check_sname(name.substr(1));
652  else {
653  std::string local_pname = pname_;
654  if(ignore_case_) {
655  local_pname = detail::to_lower(local_pname);
656  name = detail::to_lower(name);
657  }
658  if(ignore_underscore_) {
659  local_pname = detail::remove_underscore(local_pname);
660  name = detail::remove_underscore(name);
661  }
662  return name == local_pname;
663  }
664  }

667  bool check_sname(std::string name) const {
668  if(ignore_case_) { // there can be no extra underscores in check_sname
669  name = detail::to_lower(name);
670  return std::find_if(std::begin(snames_), std::end(snames_), [&name](std::string local_sname) {
671  return detail::to_lower(local_sname) == name;
672  }) != std::end(snames_);
673  } else
674  return std::find(std::begin(snames_), std::end(snames_), name) != std::end(snames_);
675  }

678  bool check_lname(std::string name) const {
679  if(ignore_case_) {
680  if(ignore_underscore_) {
681  name = detail::to_lower(detail::remove_underscore(name));
682  return std::find_if(std::begin(lnames_), std::end(lnames_), [&name](std::string local_sname) {
683  return detail::to_lower(detail::remove_underscore(local_sname)) == name;
684  }) != std::end(lnames_);
685  } else {
686  name = detail::to_lower(name);
687  return std::find_if(std::begin(lnames_), std::end(lnames_), [&name](std::string local_sname) {
688  return detail::to_lower(local_sname) == name;
689  }) != std::end(lnames_);
690  }
691 
692  } else if(ignore_underscore_) {
693  name = detail::remove_underscore(name);
694  return std::find_if(std::begin(lnames_), std::end(lnames_), [&name](std::string local_sname) {
