176 inline std::vector<std::string> split_up(std::string str) {
177 
178  const std::string delims("\'\"`");
179  auto find_ws = [](char ch) { return std::isspace<char>(ch, std::locale()); };
180  trim(str);
181 
182  std::vector<std::string> output;
183  bool embeddedQuote = false;
184  char keyChar = ' ';
185  while(!str.empty()) {
186  if(delims.find_first_of(str[0]) != std::string::npos) {
187  keyChar = str[0];
188  auto end = str.find_first_of(keyChar, 1);
189  while((end != std::string::npos) && (str[end - 1] == '\\')) { // deal with escaped quotes
190  end = str.find_first_of(keyChar, end + 1);
191  embeddedQuote = true;
192  }
193  if(end != std::string::npos) {
194  output.push_back(str.substr(1, end - 1));
195  str = str.substr(end + 1);
196  } else {
197  output.push_back(str.substr(1));
198  str = "";
199  }
200  } else {
201  auto it = std::find_if(std::begin(str), std::end(str), find_ws);
202  if(it != std::end(str)) {
203  std::string value = std::string(str.begin(), it);
204  output.push_back(value);
205  str = std::string(it, str.end());
206  } else {
207  output.push_back(str);
208  str = "";
209  }
210  }
211  // transform any embedded quotes into the regular character
212  if(embeddedQuote) {
213  output.back() = find_and_replace(output.back(), std::string("\\") + keyChar, std::string(1, keyChar));
214  embeddedQuote = false;
215  }
216  trim(str);
217  }
218  return output;
219 }

241 inline size_t escape_detect(std::string &str, size_t offset) {
242  auto next = str[offset + 1];
243  if((next == '\"') || (next == '\'') || (next == '`')) {
244  auto astart = str.find_last_of("-/ \"\'`", offset - 1);
245  if(astart != std::string::npos) {
246  if(str[astart] == (str[offset] == '=') ? '-' : '/')
247  str[offset] = ' '; // interpret this as a space so the split_up works properly
248  }
249  }
250  return offset + 1;
251 }
